1use std::str::FromStr;
2
3use crate::lyric_providers::netease::Netease;
4use crate::lyric_providers::qqmusic::QQMusic;
5
6use crate::sync::interop::mpris::PLAYER;
7use crate::sync::lyric::fetch::tricks::{get_lrc_path, lyric_tag_exists};
8use crate::sync::lyric::fetch::LyricHint;
9use crate::sync::ENABLE_LOCAL_LYRIC;
10
11pub fn hint_from_player() -> Option<LyricHint> {
12    PLAYER.with_borrow(|player| {
13        let player = player
14            .as_ref()
15            .expect("player not exists in lyric fetching");
16        let player_name = player.identity();
17        let player_bus_name = player.bus_name_player_name_part();
18        let meta = player.get_metadata().ok()?;
19
20        match (player_name, player_bus_name) {
21            ("ElectronNCM" | "Qcm", _) | (_, "musicfox" | "NeteaseCloudMusicGtk4") => meta
22                .get("mpris:trackid")
23                .and_then(mpris::MetadataValue::as_str)
24                .and_then(|s| s.split('/').last())
25                .map(str::to_owned)
26                .map(|song_id| LyricHint::SongId {
27                    song_id,
28                    provider: &Netease,
29                }),
30            ("feeluown", _) => meta
31                .url()?
32                .strip_prefix("fuo://netease/songs/")
33                .map(str::to_owned)
34                .map(|song_id| LyricHint::SongId {
35                    song_id,
36                    provider: &Netease,
37                })
38                .or_else(|| {
39                    meta.url()?
40                        .strip_prefix("fuo://qqmusic/songs/")
41                        .map(str::to_owned)
42                        .map(|song_id| LyricHint::SongId {
43                            song_id,
44                            provider: &QQMusic,
45                        })
46                }),
47            ("YesPlayMusic", _) => meta
48                .url()?
49                .strip_prefix("/trackid/")
50                .map(str::to_owned)
51                .map(|song_id| LyricHint::SongId {
52                    song_id,
53                    provider: &Netease,
54                }),
55            _ => meta.url().and_then(|meta_url| match meta_url {
56                _ if meta_url.starts_with("file://") => {
57                    if !*ENABLE_LOCAL_LYRIC
58                        .get()
59                        .expect("enable-local-lyric was not set!")
60                    {
61                        return None;
62                    }
63                    let music_path = url::Url::from_str(meta_url)
64                        .ok()
65                        .and_then(|music_uri| music_uri.to_file_path().ok())?;
66                    if lyric_tag_exists(&music_path) {
67                        return Some(LyricHint::LyricMetadata(music_path));
68                    }
69
70                    get_lrc_path(music_path)
71                        .filter(|lyric_path| lyric_path.exists())
72                        .map(LyricHint::LyricFile)
73                }
74                _ => None,
75            }),
76        }
77    })
78}
