61thread_local! {
62    static PLAYER: RefCell<Option<Player>> = const { RefCell::new(None) };
63    static PLAYER_FINDER: RefCell<PlayerFinder> = RefCell::new(PlayerFinder::new().unwrap());
64}
65
66fn find_players() -> Vec<Player> {
67    PLAYER_FINDER.with_borrow(|player_finder| match player_finder.find_all() {
68        Ok(players) => players,
69        Err(e) => {
70            error!("cannot find players!: {e}");
71            panic!("please check your d-bus connection!")
72        }
73    })
74}

76impl TryFrom<Metadata> for TrackMeta {
77    type Error = Infallible;
78
79    fn try_from(meta: Metadata) -> Result<Self, Self::Error> {
80        let track_id = meta.track_id();
81        let title = meta.title().map(str::to_string);
82        let album = meta.album_name().map(ToOwned::to_owned);
83        let artists: Option<Vec<_>> = meta
84            .artists()
85            .map(|v| v.iter().map(ToString::to_string).collect());
86        let length = meta.length();
87
88        Ok(Self {
89            unique_song_id: track_id.map(|id| id.to_string()),
90            title,
91            album,
92            artists,
93            length,
94        })
95    }
