21pub enum LyricHint {
22    SongId {
23        song_id: String,
24        provider: &'static dyn LyricProvider,
25    },
26    LyricFile(PathBuf),
27    LyricMetadata(PathBuf),
28    Metadata(TrackMeta),
29}

31pub enum LyricHintResult {
32    Lyric {
33        olyric: LyricOwned,
34        tlyric: LyricOwned,
35    },
36}

38pub async fn get_lyric_hint_from_player() -> Option<LyricHintResult> {
39    let hint_from_player: Option<LyricHint> = OS::hint_from_player();
40
41    debug!("got player hint: {:?}", hint_from_player);
42
43    match hint_from_player {
44        Some(LyricHint::SongId { song_id, provider }) => {
45            if !LYRIC_PROVIDERS.get().iter().any(|&providers| {
46                providers
47                    .iter()
48                    .any(|pro| pro.unique_name() == provider.unique_name())
49            }) {
50                warn!(
51                    "provider {} suggested by hint is not configured, skipping SongId hint",
52                    provider.unique_name()
53                );
54                return None;
55            }
56
57            crate::log::debug!("spawned query from get_accurate_lyric");
58
59            let lyric = provider.query_lyric(&song_id).await.ok()?;
60            let olyric = provider.parse_lyric(&lyric);
61            let tlyric = provider.parse_translated_lyric(&lyric);
62
63            Some(LyricHintResult::Lyric { olyric, tlyric })
64        }
65        Some(LyricHint::LyricFile(path)) => {
66            let (olyric, tlyric) = load_local_lyric(&path)?;
67            Some(LyricHintResult::Lyric { olyric, tlyric })
68        }
69        Some(LyricHint::LyricMetadata(path)) => {
70            let (olyric, tlyric) = get_lrc_from_music_metadata(&path)?;
71            Some(LyricHintResult::Lyric { olyric, tlyric })
72        }
73        _ => None,
74    }
75}

89pub static LYRIC_TAG_CACHE: Lazy<DashMap<PathBuf, bool>> = Lazy::new(DashMap::new);
90
91pub fn lyric_tag_exists(music_path: &Path) -> bool {
92    // 尝试从缓存中获取结果
93    if let Some(result) = LYRIC_TAG_CACHE.get(music_path) {
94        return *result;
95    }
96
97    let result = read_from_path(music_path)
98        .map_err(|e| warn!("cannot read music file: {e}"))
99        .ok()
100        .as_ref()
101        .and_then(|tagged_file| tagged_file.primary_tag())
102        .and_then(|tag| tag.get(&ItemKey::Lyrics))
103        .is_some();
104    // 将结果存入缓存
105    LYRIC_TAG_CACHE.insert(music_path.to_owned(), result);
106    result
107}
108pub fn get_lrc_from_music_metadata(music_path: &PathBuf) -> Option<(LyricOwned, LyricOwned)> {
109    read_from_path(music_path)
110        .map_err(|e| error!("cannot read music file: {e}"))
111        .ok()
112        .as_ref()
113        .and_then(|tagged_file| tagged_file.primary_tag())
114        .and_then(|tag| tag.get_string(&ItemKey::Lyrics))
115        .and_then(parse_local_lyric)
116}

120#[allow(unused_mut)]
121fn parse_local_lyric(lyric: &str) -> Option<(LyricOwned, LyricOwned)> {
122    let mut olyric =
123        crate::lyric_providers::utils::lrc_iter(lyric.trim_start_matches('\u{feff}').lines())
124            .map(|lyrics| Lyric::LineTimestamp(lyrics).into_owned())
125            .map_err(|e| error!("cannot parse lyric from hint: {e}"))
126            .ok()
127            .unwrap_or_default();
128    let mut tlyric = LyricOwned::None;
129    #[cfg(feature = "i18n-local-lyric")]
130    if EXTRACT_TRANSLATED_LYRIC.get().cloned().unwrap_or_default() {
131        if let LyricOwned::LineTimestamp(lines) = &olyric {
132            let tlyric_lines = extract_translated_lyric(lines);
133            if !tlyric_lines.is_empty() {
134                let olyric_lines = filter_original_lyric(lines, &tlyric_lines);
135                debug!("extracted original lyric: {olyric_lines:#?}");
136                debug!("extracted translation lyric: {tlyric_lines:#?}");
137                olyric = LyricOwned::LineTimestamp(olyric_lines);
138                tlyric = LyricOwned::LineTimestamp(tlyric_lines);
139            }
140        }
141    }
142
143    if olyric.is_none() && tlyric.is_none() {
144        return None;
145    }
146
147    Some((olyric, tlyric))
148}

150#[allow(unused_mut)]
151fn load_local_lyric<P: AsRef<Path>>(path: P) -> Option<(LyricOwned, LyricOwned)> {
152    let olyric = fs::read_to_string(&path)
153        .map_err(|e| error!("cannot read lyric from hint: {e}"))
154        .ok()
155        .as_ref()
156        .and_then(|lyric| parse_local_lyric(lyric))
157        .map(|(olyric, _)| olyric)
158        .unwrap_or_default();
159
160    #[cfg(feature = "i18n-local-lyric")]
161    let tlyric = {
162        let mut translation_path = path.as_ref().to_owned();
163        let lang = sys_locale::get_locale();
164        translation_path.set_extension(format!("{}.lrc", lang.as_deref().unwrap_or("zh")));
165        fs::read_to_string(&translation_path)
166            .map_err(|e| error!("cannot read translated lyric from hint: {e}"))
167            .ok()
168            .as_ref()
169            .and_then(|lyric| parse_local_lyric(lyric))
170            .map(|(tlyric, _)| tlyric)
171            .unwrap_or_default()
172    };
173    #[cfg(not(feature = "i18n-local-lyric"))]
174    let mut tlyric = LyricOwned::None;
175
176    if olyric.is_none() && tlyric.is_none() {
177        return None;
178    }
179
180    Some((olyric, tlyric))
181}
