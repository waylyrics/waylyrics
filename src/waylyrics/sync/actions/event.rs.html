10#[derive(Clone, Debug)]
11pub enum PlayAction {
12    Connect(String),
13    Disconnect,
14    ReloadLyric,
15    RefetchLyric,
16    RemoveLyric,
17    SearchLyric,
18    #[cfg(feature = "import-lyric")]
19    ImportOriginalLyric,
20    #[cfg(feature = "import-lyric")]
21    ImportTranslatedLyric,
22    #[cfg(feature = "export-lyric")]
23    ExportOriginalLyric,
24    #[cfg(feature = "export-lyric")]
25    ExportTranslatedLyric,
26}
27
28fn register_play_action(app: WeakRef<Application>) -> Sender<PlayAction> {
29    let (tx, rx) = async_channel::unbounded();
30
31    glib::spawn_future_local(async move {
32        while let Ok(action) = rx.recv().await {
33            debug!("Received Play Action {action:?}");
34            let (action_name, parameter) = match action {
35                PlayAction::Connect(player_id) => ("connect", Some(player_id.to_variant())),
36                PlayAction::Disconnect => ("disconnect", None),
37                PlayAction::ReloadLyric => ("reload-lyric", None),
38                PlayAction::RefetchLyric => ("refetch-lyric", None),
39                PlayAction::RemoveLyric => ("remove-lyric", None),
40                PlayAction::SearchLyric => ("search-lyric", None),
41                #[cfg(feature = "import-lyric")]
42                PlayAction::ImportOriginalLyric => ("import-lyric", Some(true.to_variant())),
43                #[cfg(feature = "import-lyric")]
44                PlayAction::ImportTranslatedLyric => ("import-lyric", Some(false.to_variant())),
45                #[cfg(feature = "export-lyric")]
46                PlayAction::ExportOriginalLyric => ("export-lyric", Some(true.to_variant())),
47                #[cfg(feature = "export-lyric")]
48                PlayAction::ExportTranslatedLyric => ("export-lyric", Some(false.to_variant())),
49            };
50
51            if let Some(app) = app.upgrade() {
52                app.activate_action(action_name, parameter.as_ref());
53            }
54        }
55    });
56
57    tx
58}
59
60pub fn init_play_action_channel(app: WeakRef<Application>) {
61    let tx = register_play_action(app);
62    PLAY_ACTION.set(tx).expect("must only initialize once");
63}
64
65pub static PLAY_ACTION: OnceLock<Sender<PlayAction>> = OnceLock::new();
