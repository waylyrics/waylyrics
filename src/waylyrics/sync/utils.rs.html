18/// both singer and album are optional non-empty string
19pub fn match_likely_lyric<'a>(
20    album: Option<&str>,
21    title: &str,
22    singer: Option<&str>,
23    length: Option<Duration>,
24    search_result: &'a [SongInfo],
25    length_toleration_ms: u128,
26) -> Option<(&'a str, u8)> {
27    length
28        .and_then(|leng| {
29            search_result
30                .iter()
31                .find(|SongInfo { length, .. }| {
32                    length.as_millis().abs_diff(leng.as_millis()) <= length_toleration_ms
33                })
34                .map(|song| (song, 1))
35        })
36        .or_else(|| {
37            // if we get only title, it is likely
38            // the player is giving messy title, which is not applicable
39            // to fuzzy-match. so we skip fuzzy-match and use
40            // the first result instead
41            if album.is_none() && singer.is_none() {
42                return None;
43            }
44
45            #[cfg(feature = "opencc")]
46            let Ok(opencc) = opencc_rust::OpenCC::new("t2s.json") else {
47                error!("opencc enabled but missing t2s.json dictionary");
48                return None;
49            };
50
51            let o_title: Vec<char>;
52            let o_album: Option<Vec<char>>;
53            let o_singer: Option<Vec<char>>;
54
55            #[cfg(feature = "opencc")]
56            {
57                o_title = opencc.convert(title).chars().collect();
58                o_album = album.map(|a| opencc.convert(a).chars().collect());
59                o_singer = singer.map(|s| opencc.convert(s).chars().collect());
60            }
61            #[cfg(not(feature = "opencc"))]
62            {
63                o_title = title.chars().collect();
64                o_album = album.map(|a| a.chars().collect());
65                o_singer = singer.map(|s| s.chars().collect());
66            }
67
68            search_result
69                .iter()
70                .map(
71                    |s @ SongInfo {
72                         title: _title,
73                         singer: _singer,
74                         album: _album,
75                         ..
76                     }| {
77                        let r_title;
78                        let r_album;
79                        let r_singer;
80
81                        #[cfg(feature = "opencc")]
82                        {
83                            r_title = opencc.convert(_title).chars().collect::<Vec<_>>();
84                            r_album = _album
85                                .as_ref()
86                                .map(|a| opencc.convert(a).chars().collect::<Vec<_>>());
87                            r_singer = opencc.convert(_singer).chars().collect::<Vec<_>>();
88                        }
89                        #[cfg(not(feature = "opencc"))]
90                        {
91                            r_title = _title.chars().collect::<Vec<_>>();
92                            r_album = _album.as_ref().map(|a| a.chars().collect::<Vec<_>>());
93                            r_singer = _singer.chars().collect::<Vec<_>>();
94                        }
95
96                        let likelihood = fuzzy_match_song(
97                            &o_title,
98                            o_album.as_deref(),
99                            o_singer.as_deref(),
100                            &r_title,
101                            r_album.as_deref(),
102                            &r_singer,
103                        );
104                        trace!("p={likelihood} for {s:?}");
105                        (s, likelihood)
106                    },
107                )
108                .max_by_key(|(_, likelihood)| (likelihood * 1024.) as u32)
109                .map(|(s, _)| (s, 0))
110        })
111        .or(search_result.first().map(|song| (song, 2)))
112        .map(|(song, weight)| (song.id.as_str(), weight))
113}

123pub fn fuzzy_match_song(
124    title: &[char],
125    album: Option<&[char]>,
126    singer: Option<&[char]>,
127    r_title: &[char],
128    r_album: Option<&[char]>,
129    r_singer: &[char],
130) -> f64 {
131    let title_likelihood = distance(title, r_title);
132    let singer_likelihood = || distance(singer.unwrap_or_default(), r_singer);
133    let album_likelihood = || distance(album.unwrap_or_default(), r_album.unwrap_or_default());
134    match (singer, album) {
135        (Some(_), Some(_)) => {
136            title_likelihood * 0.4 + singer_likelihood() * 0.2 + album_likelihood() * 0.4
137        }
138        (Some(_), None) => title_likelihood * 0.7 + singer_likelihood() * 0.3,
139        (None, Some(_)) => title_likelihood * 0.8 + album_likelihood() * 0.2,
140        (None, None) => title_likelihood,
141    }
142}

144pub fn extract_translated_lyric(lyric: impl AsRef<[LyricLineOwned]>) -> Vec<LyricLineOwned> {
145    let tlyric_lines = lyric
146        .as_ref()
147        .windows(2)
148        .filter_map(|l| <&[LyricLineOwned; 2]>::try_from(l).ok())
149        // this should work because we have sorted lyrics by timestamp
150        .filter(|&[a, b]| a.start_time == b.start_time)
151        .map(|p| &p[1])
152        .cloned()
153        .collect::<Vec<_>>();
154    tlyric_lines
155}

157pub fn filter_original_lyric(
158    lyric: impl AsRef<[LyricLineOwned]>,
159    tlyric: impl AsRef<[LyricLineOwned]>,
160) -> Vec<LyricLineOwned> {
161    let tlyric = tlyric
162        .as_ref()
163        .iter()
164        .map(|LyricLineOwned { text, start_time }| (start_time, text))
165        .collect::<HashMap<_, _>>();
166    let tlyric_lines = lyric
167        .as_ref()
168        .iter()
169        .filter(|LyricLineOwned { text, start_time }| {
170            !tlyric.contains_key(start_time) || tlyric[start_time] != text
171        })
172        .cloned()
173        .collect::<Vec<_>>();
174    tlyric_lines
175}
