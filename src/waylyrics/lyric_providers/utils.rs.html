6/// parses lrc tags in each line
7///
8/// ignores lines not started with '['
9///
10/// returned `Vec` is sorted by it's Duration
11///
12/// In the case multiple lyrics comes with the same start_time,
13///
14/// they are not reordered between them
15pub fn lrc_iter<'a>(
16    lyric_lines: impl Iterator<Item = &'a str>,
17) -> Result<Vec<LyricLine<'a>>, LrcParseError> {
18    let filtered_lines = lyric_lines.filter(|l| l.starts_with('['));
19    let mut lrc_vec: Vec<_> = filtered_lines
20        .enumerate()
21        .filter_map(|(line_num, line)| parse_single(line, line_num).ok())
22        .flatten()
23        .filter_map(|lrc_item| match lrc_item {
24            lrc_nom::LrcItem::Metadata(_) => None,
25            lrc_nom::LrcItem::Lyric(lyric, timestamps) => {
26                Some(timestamps.into_iter().map(|timestamp| LyricLine {
27                    text: lyric.trim(),
28                    start_time: Duration::from_millis(timestamp as _),
29                }))
30            }
31        })
32        .flatten()
33        .collect();
34    // handling malformed LRC timestamp by sorting them here
35    //
36    // DO NOT use sort_unstable_by_key, we should let lyrics
37    // preserve original relative order, that extracting
38    // translation will work.
39    lrc_vec.sort_by_key(|line| line.start_time);
40
41    Ok(lrc_vec)
42}

44pub fn find_next_lyric<'a>(
45    elapsed: &Duration,
46    lyric: &'a [LyricLineOwned],
47) -> Option<&'a LyricLineOwned> {
48    lyric
49        .iter()
50        .take_while(
51            |LyricLineOwned {
52                 start_time: off, ..
53             }| off <= elapsed,
54        )
55        .last()
56}
