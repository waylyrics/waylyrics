78impl Lyric<'_> {
79    pub fn into_owned(self) -> LyricOwned {
80        match self {
81            Lyric::None => LyricOwned::None,
82            Lyric::NoTimestamp => LyricOwned::NoTimestamp,
83            Lyric::LineTimestamp(lyrics) => LyricOwned::LineTimestamp(
84                lyrics
85                    .into_iter()
86                    .map(LyricLine::<'_>::into_owned)
87                    .collect(),
88            ),
89        }
90    }
91}
92
93impl LyricLine<'_> {
94    pub fn into_owned(Self { text, start_time }: Self) -> LyricLineOwned {
95        LyricLineOwned {
96            text: text.into(),
97            start_time,
98        }
99    }
100}
