30    async fn search_song_detailed(
31        &self,
32        album: &str,
33        artists: &[&str],
34        title: &str,
35    ) -> Result<Vec<super::SongInfo>> {
36        let artists = artists
37            .iter()
38            .map(|&a| *ARTIST_ALIAS_MAP.get(&*a.to_lowercase()).unwrap_or(&a))
39            .collect::<Vec<&str>>();
40        let keyword = default_search_query(album, &artists, title);
41        self.search_song(&keyword).await
42    }

44    async fn query_lyric(&self, id: &str) -> Result<LyricStore> {
45        let id = id.to_owned();
46        tokio_spawn!(async move {
47            let api = NcmApi::new(false, "");
48            let id = id.parse()?;
49            let query_result = api.lyric(id).await?;
50
51            let lyric_resp: LyricResp = query_result.deserialize()?;
52
53            crate::log::debug!("lyric query result: {lyric_resp:?}");
54
55            Ok(LyricStore {
56                lyric: lyric_resp.lrc.map(|l| l.lyric),
57                tlyric: lyric_resp.tlyric.map(|l| l.lyric),
58            })
59        })
60        .await?
61    }
62
63    async fn search_song(&self, keyword: &str) -> Result<Vec<super::SongInfo>> {
64        let keyword = keyword.to_owned();
65        tokio_spawn!(async move {
66            crate::log::debug!("search keyword: {keyword}");
67
68            let api = NcmApi::new(false, "");
69            let search_result = api.search(&keyword, None).await?;
70            let resp: SearchSongResp = search_result.deserialize()?;
71            crate::log::debug!("search result: {resp:?}");
72
73            Ok(resp
74                .result
75                .ok_or(super::Error::NoResult)?
76                .songs
77                .iter()
78                .map(
79                    |Song {
80                         name,
81                         id,
82                         artists,
83                         duration,
84                         album: Album { name: album, .. },
85                         ..
86                     }| super::SongInfo {
87                        id: id.to_string(),
88                        title: name.into(),
89                        album: album.clone(),
90                        singer: artists
91                            .iter()
92                            .filter_map(|Artist { name, .. }| name.as_ref())
93                            .fold(String::default(), |mut s, op| {
94                                if !s.is_empty() {
95                                    s.push(',')
96                                }
97                                s += op;
98                                s
99                            }),
100                        length: Duration::from_millis(*duration as _),
101                    },
102                )
103                .collect())
104        })
105        .await?
106    }
107

113impl super::LyricParse for Netease {
114    fn parse_lyric(&self, store: &LyricStore) -> LyricOwned {
115        let lyric = store.lyric.as_deref();
116        verify_lyric(lyric).into_owned()
117    }
118
119    fn parse_translated_lyric(&self, store: &LyricStore) -> LyricOwned {
120        let olyric = self.parse_lyric(store);
121        let tlyric = store.tlyric.as_deref();
122        let mut tlyric = verify_lyric(tlyric).into_owned();
123
124        // workaround: hide translation if original line ends
125        if let (LyricOwned::LineTimestamp(tlyric), LyricOwned::LineTimestamp(olyric)) =
126            (&mut tlyric, &olyric)
127        {
128            let last_t = tlyric.last();
129            if let Some(next_line_time) = last_t.and_then(|t| {
130                olyric
131                    .iter()
132                    .find(|o| o.start_time > t.start_time)
133                    .map(|o| o.start_time)
134            }) {
135                tlyric.push(LyricLineOwned {
136                    text: String::default(),
137                    start_time: next_line_time,
138                });
139            }
140        }
141        tlyric
142    }
143}

145fn verify_lyric(lyric: Option<&str>) -> Lyric<'_> {
146    match lyric {
147        Some("") | None => super::Lyric::None,
148        Some(lyric) => {
149            if let Ok(parsed) = super::utils::lrc_iter(lyric.lines()) {
150                Lyric::LineTimestamp(parsed)
151            } else {
152                Lyric::NoTimestamp
153            }
154        }
155    }
