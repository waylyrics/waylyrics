7#[cfg(target_os = "windows")]
8pub(super) fn set_click_pass_through(window: &window::Window, enabled: bool) {
9    fn set_window_click_through(hwnd: isize, enabled: bool) {
10        use windows::Win32::Foundation::HWND;
11        use windows::Win32::UI::WindowsAndMessaging::{
12            GetWindowLongPtrW, SetWindowLongPtrW, GWL_EXSTYLE,
13        };
14        let hwnd = HWND(hwnd as _);
15
16        const WS_EX_TRANSPARENT: isize = 0x00000020;
17        const WS_EX_LAYERED: isize = 0x00080000;
18        unsafe {
19            let ex_style = GetWindowLongPtrW(hwnd, GWL_EXSTYLE);
20            if enabled {
21                SetWindowLongPtrW(
22                    hwnd,
23                    GWL_EXSTYLE,
24                    ex_style | WS_EX_TRANSPARENT | WS_EX_LAYERED,
25                );
26            } else {
27                SetWindowLongPtrW(
28                    hwnd,
29                    GWL_EXSTYLE,
30                    ex_style & !WS_EX_TRANSPARENT & !WS_EX_LAYERED,
31                );
32            }
33        }
34    }
35
36    let Some(surface) = window.surface().and_downcast::<gdk4_win32::Win32Surface>() else {
37        return;
38    };
39
40    let handle = surface.handle().0;
41
42    set_window_click_through(handle, enabled);
43}

45#[cfg(not(target_os = "windows"))]
46pub(super) fn set_click_pass_through(window: &window::Window, enabled: bool) {
47    use gtk::{
48        cairo::{RectangleInt, Region},
49        subclass::prelude::*,
50    };
51
52    let obj = window;
53    let Some(surface) = obj.surface() else {
54        return;
55    };
56
57    if enabled {
58        if !window.is_decorated() {
59            surface.set_input_region(&Region::create_rectangle(&RectangleInt::new(0, 0, 0, 0)));
60        } else {
61            let headerbar = &window.imp().headerbar;
62            let allocation = headerbar.allocation();
63
64            surface.set_input_region(&Region::create_rectangle(&RectangleInt::new(
65                allocation.x(),
66                allocation.y(),
67                allocation.width(),
68                allocation.height(),
69            )));
70        }
71    } else {
72        surface.set_input_region(&Region::create_rectangle(&RectangleInt::new(
73            0,
74            0,
75            i32::MAX,
76            i32::MAX,
77        )));
78    }
79}

81/// set css style for waylyrics
82/// As said in [GTK+ doc], gtk constructs style from the lower priority ones to the upper ones,
83/// We set priority as `STYLE_PROVIDER_PRIORITY + 1` to override user theme
84///
85/// [GTK+ doc]: https://docs.gtk.org/gtk4/type_func.StyleContext.add_provider_for_display.html#parameters
86pub fn merge_css(css: &str) {
87    use gtk::gdk::Display as GdkDisplay;
88    use gtk::CssProvider;
89    use std::cell::RefCell;
90
91    thread_local! {
92        static LATEST_PROVIDER: RefCell<Option<CssProvider>> = const { RefCell::new(None) };
93    }
94    let css_provider = CssProvider::new();
95    css_provider.load_from_data(css);
96    let display = GdkDisplay::default().expect("Could not connect to a display.");
97    LATEST_PROVIDER.with_borrow_mut(|provider| {
98        if let Some(provider) = provider.take() {
99            gtk::style_context_remove_provider_for_display(&display, &provider);
100        }
101    });
102
103    gtk::style_context_add_provider_for_display(
104        &display,
105        &css_provider,
106        gtk::STYLE_PROVIDER_PRIORITY_USER + 1,
107    );
108    LATEST_PROVIDER.with_borrow_mut(|provider| {
109        *provider = Some(css_provider);
110    });
111}

113fn has_filtered_word(text: &str) -> bool {
114    EXCLUDED_REGEXES.with_borrow(|regex_set| regex_set.is_match(text))
115}

117pub fn setup_label(label: &Label, hide_filtered_words: bool) {
118    label.set_wrap(true);
119    label.set_wrap_mode(gtk::pango::WrapMode::Word);
120
121    if hide_filtered_words {
122        label.connect_label_notify(|label| {
123            let text = label.label();
124            let visible = !has_filtered_word(&text) && !text.is_empty();
125            label.set_visible(visible);
126        });
127    } else {
128        label.connect_label_notify(|label| {
129            let visible = !label.label().is_empty();
130            label.set_visible(visible);
131        });
132    }
133}
