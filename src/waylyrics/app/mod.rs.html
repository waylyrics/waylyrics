18pub fn build_main_window(
19    app: &Application,
20    enable_filter_regex: bool,
21    cache_lyrics: bool,
22    length_toleration_ms: u128,
23    show_default_text_on_idle: bool,
24    show_lyric_on_pause: bool,
25    #[cfg(feature = "layer-shell")] layer_shell: bool,
26    #[cfg(feature = "layer-shell")] layer_shell_anchor: crate::config::LayerShellAnchor,
27) -> Window {
28    let window = Window::new(
29        app,
30        cache_lyrics,
31        length_toleration_ms,
32        show_default_text_on_idle,
33        show_lyric_on_pause,
34    );
35
36    #[cfg(feature = "layer-shell")]
37    if layer_shell {
38        use gtk4_layer_shell::{KeyboardMode, Layer, LayerShell};
39
40        LayerShell::init_layer_shell(&window);
41        // wlr-layer-shell-unstable-v1 version 4 needed, see
42        // https://wayland.app/protocols/wlr-layer-shell-unstable-v1#compositor-support
43        // and https://wayland.app/protocols/wlr-layer-shell-unstable-v1#zwlr_layer_surface_v1:enum:keyboard_interactivity
44        LayerShell::set_keyboard_mode(&window, KeyboardMode::OnDemand);
45        LayerShell::set_layer(&window, Layer::Overlay);
46
47        layer_shell_anchor.setup(&window);
48    }
49
50    window.set_size_request(500, WINDOW_MIN_HEIGHT);
51    window.set_title(Some(DEFAULT_TEXT));
52    window.set_icon_name(Some(crate::APP_ID_FIXED));
53    window.present();
54
55    let above_label = Label::builder()
56        .label("Waylyrics")
57        .name("above")
58        .vexpand(true)
59        .hexpand(false)
60        .build();
61    let below_label = Label::builder()
62        .label("")
63        .name("below")
64        .vexpand(true)
65        .hexpand(false)
66        .visible(false)
67        .build();
68
69    for label in [&above_label, &below_label] {
70        utils::setup_label(label, enable_filter_regex);
71    }
72
73    let verical_box = gtk::Box::builder()
74        .name("lyrics-box")
75        .baseline_position(gtk::BaselinePosition::Center)
76        .orientation(gtk::Orientation::Vertical)
77        .halign(gtk::Align::Center)
78        .valign(gtk::Align::Center)
79        .vexpand(true)
80        .hexpand(false)
81        .build();
82
83    verical_box.insert_child_after(&above_label, gtk::Box::NONE);
84    verical_box.insert_child_after(&below_label, Some(&above_label));
85
86    window.set_child(Some(&verical_box));
87
88    let align = window.imp().lyric_align.get();
89    set_lyric_align(&window, align);
90
91    window.connect_decorated_notify(|window| {
92        crate::log::debug!("triggered decorated signal");
93        let clickthrough = window.imp().clickthrough.get();
94        set_click_pass_through(window, clickthrough)
95    });
96
97    window.set_icon_name(Some(crate::APP_ID_FIXED));
98
99    window
100}

102pub fn set_lyric_align(window: &Window, align: config::Align) -> Option<()> {
103    let vbox: gtk::Box = window.child()?.downcast().ok()?;
104    vbox.set_halign(align.into());
105
106    let labels = get_labels(&vbox)?;
107    for label in labels {
108        label.set_halign(align.into());
109        label.set_justify(align.into());
110    }
111
112    window.imp().lyric_align.set(align);
113    Some(())
114}

116fn get_labels(vbox: &gtk::Box) -> Option<[Label; 2]> {
117    let above_label: Label = vbox.first_child()?.downcast().ok()?;
118    let below_label: Label = vbox.last_child()?.downcast().ok()?;
119    Some([above_label, below_label])
120}

122pub fn get_label(window: &Window, position: &str) -> Label {
123    let Some(vbox) = window.child().and_then(|c| c.downcast::<gtk::Box>().ok()) else {
124        panic!("initialization failed!")
125    };
126    get_labels(&vbox)
127        .expect("cannot find labels")
128        .into_iter()
129        .find(|label| label.widget_name() == position)
130        .unwrap()
131}
