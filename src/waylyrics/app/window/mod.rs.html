10use crate::APP_ID_FIXED;

12glib::wrapper! {
13    pub struct Window(ObjectSubclass<imp::Window>)
14        @extends gtk::ApplicationWindow, gtk::Window, gtk::Widget,
15        @implements gio::ActionGroup, gio::ActionMap, gtk::Accessible, gtk::Buildable,
16                    gtk::ConstraintTarget, gtk::Native, gtk::Root, gtk::ShortcutManager;
17}
18
19impl Window {
20    pub fn new(
21        app: &Application,
22        cache_lyrics: bool,
23        length_toleration_ms: u128,
24        show_default_text_on_idle: bool,
25        show_lyric_on_pause: bool,
26    ) -> Self {
27        let window: Self = Object::builder().property("application", app).build();
28        let imp = window.imp();
29
30        window.set_widget_name("main-window");
31
32        imp.cache_lyrics.set(cache_lyrics);
33        imp.length_toleration_ms.set(length_toleration_ms);
34        imp.show_default_text_on_idle.set(show_default_text_on_idle);
35        imp.show_lyric_on_pause.set(show_lyric_on_pause);
36
37        window
38    }
39
40    fn setup_settings(&self) {
41        let settings = Settings::new(APP_ID_FIXED);
42        self.imp()
43            .settings
44            .set(settings)
45            .expect("`settings` should not be set before calling `setup_settings`.");
46    }
47
48    fn settings(&self) -> &Settings {
49        self.imp()
50            .settings
51            .get()
52            .expect("`settings` should be set in `setup_settings`.")
53    }
54
55    pub fn save_window_state(&self) -> Result<(), glib::BoolError> {
56        let (width, height) = self.default_size();
57        let decorated = self.is_decorated();
58        let click_through = self.imp().clickthrough.get();
59        let align_mode = self.imp().lyric_align.get().to_string();
60        let display_mode = self.imp().lyric_display_mode.get().to_string();
61
62        self.settings().set_int("window-width", width)?;
63        self.settings().set_int("window-height", height)?;
64        self.settings().set_boolean("window-decorated", decorated)?;
65        self.settings()
66            .set_boolean("window-click-through", click_through)?;
67        self.settings()
68            .set_string("lyric-align-mode", &align_mode)?;
69        self.settings()
70            .set_string("lyric-display-mode", &display_mode)?;
71
72        Ok(())
73    }
74
75    fn load_window_state(&self) {
76        let height = self.settings().int("window-height");
77        let width = self.settings().int("window-width");
78        let decorated = self.settings().boolean("window-decorated");
79        let click_through = self.settings().boolean("window-click-through");
80        let align_mode: Align = self.settings().string("lyric-align-mode").parse().unwrap();
81        let display_mode: LyricDisplayMode = self
82            .settings()
83            .string("lyric-display-mode")
84            .parse()
85            .unwrap();
86
87        self.set_default_size(width, height);
88        self.set_decorated(decorated);
89        self.imp().clickthrough.set(click_through);
90        self.imp().lyric_align.set(align_mode);
91        self.imp().lyric_display_mode.set(display_mode)
92    }
93}
