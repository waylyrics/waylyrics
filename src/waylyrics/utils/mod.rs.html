30pub fn parse_time(time: &str) -> Result<Duration, ParseError> {
31    use rust_decimal::prelude::*;
32    use rust_decimal_macros::dec;
33
34    let time_ms = if time.ends_with("ms") {
35        let sec = time.trim_end_matches("ms");
36        Decimal::from_str_exact(sec)?
37    } else if time.ends_with('s') {
38        let milli_sec = time.trim_end_matches('s');
39        Decimal::from_str_exact(milli_sec)? * dec!(1000)
40    } else {
41        return Err(ParseError::IllFormed);
42    };
43
44    Ok(Duration::from_millis(
45        time_ms.to_u64().ok_or(ParseError::ExceedsLimits)?,
46    ))
47}

61pub fn init_dirs() -> Result<(PathBuf, PathBuf)> {
62    let proj_dirs =
63        directories::ProjectDirs::from("io", "poly000", "waylyrics").expect("can't get proj_dirs!");
64    let config_home = proj_dirs.config_dir();
65    let cache_dir = proj_dirs.cache_dir();
66
67    crate::CACHE_DIR.set(
68        cache_dir
69            .to_str()
70            .expect("xdg config home is not valid UTF-8")
71            .into(),
72    );
73
74    std::fs::create_dir_all(config_home)?;
75    std::fs::create_dir_all(cache_dir)?;
76    let config_path = config_home.join("config.toml");
77    let user_theme_dir = proj_dirs.data_dir().join("_themes");
78
79    if !config_path.exists() {
80        std::fs::write(&config_path, toml::to_string(&Config::default())?)?;
81    }
82    if !user_theme_dir.exists() {
83        std::fs::create_dir_all(&user_theme_dir)?;
84    }
85
86    Ok((config_path, user_theme_dir))
87}
