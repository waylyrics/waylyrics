7pub fn auto_theme_change(color_scheme: ColorScheme, theme_switch: bool) {
8    let Some(settings) = gtk::Settings::default() else {
9        return;
10    };
11
12    match color_scheme {
13        ColorScheme::Light => settings.set_gtk_application_prefer_dark_theme(false),
14        ColorScheme::Dark => settings.set_gtk_application_prefer_dark_theme(true),
15        ColorScheme::Auto => {
16            let theme = dark_light::sync::detect();
17            info!("detected theme: {theme:?}");
18            if theme == dark_light::Mode::Dark {
19                settings.set_gtk_application_prefer_dark_theme(true);
20                if theme_switch {
21                    set_and_update(true);
22                }
23            } else {
24                settings.set_gtk_application_prefer_dark_theme(false)
25            }
26
27            // Listen to changes...
28            glib_spawn!(async move {
29                let mut stream = dark_light::subscribe().await;
30                while let Some(mode) = stream.next().await {
31                    set_by_mode(mode, theme_switch, &settings);
32                }
33            });
34        }
35    }
36}

39fn replace_suffix(input: &str, old_suffix: &str, new_suffix: &str) -> String {
40    if let Some(trimmed) = input.strip_suffix(old_suffix) {
41        format!("{}{}", trimmed, new_suffix)
42    } else {
43        input.to_string()
44    }
45}

47fn set_and_update(dark: bool) {
48    THEME_PATH.with_borrow_mut(|theme_path| {
49        let filename: &str = match theme_path.file_name().and_then(|p| p.to_str()) {
50            Some(p) => p,
51            None => return,
52        };
53        if dark {
54            if !filename.ends_with("-dark.css") {
55                let new_name = replace_suffix(filename, ".css", "-dark.css");
56                theme_path.set_file_name(new_name);
57            }
58        } else if filename.ends_with("-dark.css") {
59            let new_name = replace_suffix(filename, "-dark.css", ".css");
60            theme_path.set_file_name(new_name);
61        }
62
63        if let Ok(style) = std::fs::read_to_string(&theme_path) {
64            crate::app::utils::merge_css(&style);
65        } else {
66            log::warn!("Filename {:?} not found.", theme_path);
67        }
68    })
69}

71fn set_by_mode(mode: dark_light::Mode, theme_switch: bool, settings: &gtk::Settings) {
72    let is_dark_mode = mode == dark_light::Mode::Dark;
73
74    settings.set_gtk_application_prefer_dark_theme(is_dark_mode);
75    if theme_switch {
76        set_and_update(is_dark_mode);
77    }
