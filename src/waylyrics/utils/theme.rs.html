6pub fn auto_theme_change(color_scheme: ColorScheme, theme_switch: bool) {
7    let Some(settings) = gtk::Settings::default() else {
8        return;
9    };
10
11    match color_scheme {
12        ColorScheme::Light => settings.set_gtk_application_prefer_dark_theme(false),
13        ColorScheme::Dark => settings.set_gtk_application_prefer_dark_theme(true),
14        ColorScheme::Auto => {
15            let theme = dark_light::sync::detect();
16            info!("detected theme: {theme:?}");
17            if theme == dark_light::Mode::Dark {
18                settings.set_gtk_application_prefer_dark_theme(true);
19                if theme_switch {
20                    set_and_update(true);
21                }
22            } else {
23                settings.set_gtk_application_prefer_dark_theme(false)
24            }
25
26            // Listen to changes...
27            glib_spawn!(async move {
28                let mut stream = dark_light::subscribe().await;
29                while let Some(mode) = stream.next().await {
30                    set_by_mode(mode, theme_switch, &settings);
31                }
32            });
33        }
34    }
35}

38fn replace_suffix(input: &str, old_suffix: &str, new_suffix: &str) -> String {
39    if let Some(trimmed) = input.strip_suffix(old_suffix) {
40        format!("{}{}", trimmed, new_suffix)
41    } else {
42        input.to_string()
43    }
44}

46fn set_and_update(dark: bool) {
47    THEME_PATH.with_borrow_mut(|theme_path| {
48        let filename: &str = match theme_path.file_name().and_then(|p| p.to_str()) {
49            Some(p) => p,
50            None => return,
51        };
52        if dark {
53            if !filename.ends_with("-dark.css") {
54                let new_name = replace_suffix(filename, ".css", "-dark.css");
55                theme_path.set_file_name(new_name);
56            }
57        } else if filename.ends_with("-dark.css") {
58            let new_name = replace_suffix(filename, "-dark.css", ".css");
59            theme_path.set_file_name(new_name);
60        }
61
62        if let Ok(style) = std::fs::read_to_string(&theme_path) {
63            crate::app::utils::merge_css(&style);
64        } else {
65            log::warn!("Filename {:?} not found.", theme_path);
66        }
67    })
68}

70fn set_by_mode(mode: dark_light::Mode, theme_switch: bool, settings: &gtk::Settings) {
71    let is_dark_mode = mode == dark_light::Mode::Dark;
72
73    settings.set_gtk_application_prefer_dark_theme(is_dark_mode);
74    if theme_switch {
75        set_and_update(is_dark_mode);
76    }
